\begin{frame}<0>[label=RISCattrs]{typical RISC ISA properties}
    \begin{itemize}
        \item fewer, simpler instructions
        \item \myemph<2>{seperate instructions to access memory}
        \item \sout<2>{fixed-length instructions}
        \item more registers
        \item \myemph<2>{no ``loops'' within single instructions}
        \item \myemph<2>{no instructions with two memory operands}
        \item \myemph<2>{few addressing modes}
    \end{itemize}
\end{frame}
