\begin{frame}[fragile,label=muxExercise]{MUX exercise}
\begin{Verbatim}
foo = [
    bar > 10 : 100;
    (bar & 1) == 1 : 200;
    bar < 20 : 300;
    1 : 400;
]
\end{Verbatim}
exercise 1: if \texttt{bar} is 9, what is foo? \\
exercise 2: if \texttt{bar} is 10, what is foo? \\
exercise 3: if \texttt{bar} is 11, what is foo? \\
\end{frame}
