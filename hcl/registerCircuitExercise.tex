\begin{frame}[fragile,label=hclRegCircuitEx]{HCL circuit with registers}
\begin{Verbatim}
register xY {
    a : 4 = 1;  /* <-- initial Y_a */
    b : 4 = 1;  /* <-- initial Y_b */
}
x_b = x_a + Y_a;
x_a = Y_a + Y_b;
\end{Verbatim}
% zero: x_a = 2, x_b = 3
% one cycle: Y_a = 2, Y_b = 3; x_a = 5, x_b = 7
% two cycles Y_a = 5, Y_b = 7
