\begin{frame}[fragile,label=hcl2DPipe]{HCLRS pipeline registers}
\begin{Verbatim}[fontsize=\fontsize{10}{11}\selectfont]
register pP {
    pc : 64 = 0;
};
/* Fetch+PC Update*/
register fD {
    rA : 4 = REG_NONE; rB : 4 = REG_NONE;
};
/* Decode */
register dE {
    valA : 64 = 0; valB : 64 = 0; dstE : 4 = REG_NONE;
}
/* Execute */
register eW {
    valE : 64 = 0; dstE : 4 = REG_NONE;
}
/* Writeback */
\end{Verbatim}
\end{frame}
