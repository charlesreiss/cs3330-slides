\begin{frame}<9>[label=exceptTypes]{types of exceptions}
\begin{itemize}
\item \myemph<1,5>{interrupts} --- externally-triggered
    \begin{itemize}
    \item \myemph<1>{timer} --- keep program from hogging CPU
    \item \myemph<5>{I/O devices} --- key presses, hard drives, networks, \ldots
    \end{itemize}
\item \myemph<3-4>{traps} --- intentionally triggered exceptions
    \begin{itemize}
    \item \myemph<3-4>{system calls} --- ask OS to do something
    \end{itemize}
\item \myemph<2,7,8>{faults} --- errors/events in programs
    \begin{itemize}
        \item \myemph<8>{memory not in address space} (``Segmentation fault'')
        \item \myemph<7>{privileged instruction}
    \item divide by zero
    \item invalid instruction
    \end{itemize}
\item \myemph<6>{aborts} --- hardware is broken
% FIXME: aborts
\end{itemize}
\end{frame}
