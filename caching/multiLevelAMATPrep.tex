\begin{frame}{AMAT example}
    \begin{itemize}
    \item suppose cache has 10 cycle hit time
    \item 80\% hit rate
    \item 100 cycle miss penalty
    \vspace{.5cm}
    \item<2-> AMAT = hit time + miss rate $\times$ miss penalty
    \item<2-> AMAT = 10 cycles + (100\% - 80\%) $\times$ 100 cycles = 30 cycles
    \end{itemize}
\end{frame}
