\begin{frame}{protection and sudo}
\begin{itemize}
    \item programs \myemph{always} run in user mode
    \item extra permissions from OS \myemph{do not change this}
        \begin{itemize}
        \item sudo, superuser, root, SYSTEM, \ldots
        \end{itemize}
    \item \myemph{operating system} may remember extra privileges
\end{itemize}
\end{frame}
