\begin{frame}{powers of two: forward}
\begin{itemize}
    \item<1-> $2^{35}$ \onslide<2->{$=2^5 \cdot 2^{30} = 32G$ (30 = G)}
    \item<1-> $2^{21}$ \onslide<4->{$=2^1 \cdot 2^{20} = 2M$ (20 = M)}
    \item<1-> $2^{9}$ \onslide<5->{$= 512$}
    \item<1-> $2^{14}$ \onslide<6->{$=2^4 \cdot 2^{10} = 16K$}
\end{itemize}
\end{frame}

\begin{frame}{powers of two: backward}
\begin{itemize}
    \item<1-> 16G \onslide<2->{$=16 \cdot 2^{30} = 2^{30 + 4} = 2^{34}$}
    \item<1-> 128K \onslide<3->{$=128 \cdot 2^{10} = 2^{10 + 7} = 2^{17}$}
    \item<1-> 4M \onslide<4->{$=4 \cdot 2^{20} = 2^{20 + 2} = 2^{22}$}
    \item<1-> 256T \onslide<4->{$=256 \cdot 2^{40} = 2^{40 + 8} = 2^{48}$}
\end{itemize}
\end{frame}
