\begin{frame}{using YAS}
    \begin{itemize}
    \item download HCLRS (we'll use it later)
    \item extract the archive
    \item run \texttt{make}
    \end{itemize}
\end{frame}

\begin{frame}[fragile,label=exampleYs]{example.ys}
    \begin{itemize}
    \item example.ys:
    \end{itemize}
\begin{Verbatim}
    irmovq $100, %rax
    irmovq $1, %rcx
    irmovq $10, %rdx
loop:
    subq %rdx, %rax 
    subq %rcx, %rdx
    jg loop
    halt
\end{Verbatim}
\end{frame}
